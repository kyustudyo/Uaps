🔧 핵심 코드

// UnitSO.cs - ScriptableObject 클래스 using UnityEngine; namespace GameDevTV.RTS.Units { [CreateAssetMenu(fileName = "Unit", menuName = "Units/Unit")] public class UnitSO : ScriptableObject { [field: SerializeField] public int Health { get; private set; } = 100; } }
// AbstractCommandable.cs - 수정된 부분 public abstract class AbstractCommandable : MonoBehaviour, ISelectable { [field: SerializeField] public int CurrentHealth { get; private set; } [field: SerializeField] public int MaxHealth { get; private set; } [SerializeField] private DecalProjector decalProjector; [SerializeField] private UnitSO UnitSO; protected virtual void Start() { CurrentHealth = UnitSO.Health; MaxHealth = UnitSO.Health; } // Select/Deselect 메서드는 동일... }
// AbstractUnit.cs - Start() 오버라이드 public abstract class AbstractUnit : AbstractCommandable, IMoveable { public float AgentRadius => agent.radius; private NavMeshAgent agent; protected virtual void Awake() { agent = GetComponent(); } protected override void Start() { base.Start(); // 부모 클래스의 Start() 호출 Bus.Raise(new UnitSpawnEvent(this)); } public void MoveTo(Vector3 position) { agent.SetDestination(position); } }

❓ 왜 필요한가?

📋 ScriptableObject의 장점

설정 데이터 분리: 코드와 데이터를 명확히 분리

디자이너 친화적: 프로그래머 없이도 설정 변경 가능

재사용성: 하나의 설정으로 여러 Prefab에 적용

버전 관리: 설정 변경사항을 Git으로 추적 가능

🔄 MonoBehaviour vs ScriptableObject

MonoBehaviour: GameObject에 부착, 런타임 데이터 관리

ScriptableObject: 프로젝트 파일에 저장, 빌드 시점 설정

역할 분리: 각각의 고유한 역할을 명확히 구분

⚠️ 주의사항

빌드 시점 데이터: 런타임에 변경된 데이터는 저장되지 않음

에디터 vs 플레이어: 에디터에서는 변경사항이 유지되지만 플레이어에서는 초기화

상속 처리: Start() 메서드 오버라이드 시 base.Start() 호출 필수

  • 리소스 설정: 각 리소스의 수집량, 재생 속도를 ScriptableObject로 관리
